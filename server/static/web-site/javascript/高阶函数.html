                                            function deBounce(fn: (...arg: any[]) => any, delay = 500) {
  let time: NodeJS.Timeout = null
  return function (...args: any[]) {
    time && clearTimeout(time)
    time = setTimeout(() => {
      fn(...args)
    }, delay)
  }
}

                                            function throttle(fn: (...arg: any[]) => any, delay = 500) {
  let flag = true
  return function (...args: any[]) {
    if (!flag) return
    flag = false
    setTimeout(() => {
      flag = true
      fn(...args)
    }, delay)
  }
}
