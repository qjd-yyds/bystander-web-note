# 高阶函数

  • 高阶函数是对其他函数进行操作的函数，可以将它们作为参数或返回他们
  • 简单来说，高阶函数就是一个函数，他接受函数作为参数或将函数作为输出返回

# reduce

TIP

会创建一个新的数组

对一个数组进行求和

const arr = [1, 2, 3, 4, 5]
const sum = arr.reduce((prev, cur) => {
  return prev + cur
}, 0)
console.log(sum) // 15

# compose 函数

  • 将需要嵌套执行的函数平铺
  • 嵌套执行指的是，一个函数的返回值作为下一个函数的参数
  • 实现函数式编程中的pointfree，使我们专注于转换而不是数据
function Compose() {
  var fns = Array.prototype.slice.call(arguments)
  return function (x) {
    return fns.reduceRight((res, fn) => {
      return fn(res)
    }, x)
  }
}

# Pipe 函数

和 compose 函数一样，不过是从左往右执行

function Pipe() {
  var fns = Array.prototype.slice.call(arguments)
  return function (x) {
    return fns.reduce((res, fn) => {
      return fn(res)
    }, x)
  }
}

# 防抖(deBounce)

不防抖
防抖
事件相应函数在一段时间后才执行，如果在这段事件内再次调用，则重新计算执行时间 
function deBounce(fn: (...arg: any[]) => any, delay = 500) {
  let time: NodeJS.Timeout = null
  return function (...args: any[]) {
    time && clearTimeout(time)
    time = setTimeout(() => {
      fn(...args)
    }, delay)
  }
}

# 节流(throttle)

不节流
节流
当持续触发事件的时候，保证一定时间段内只调用一次 
function throttle(fn: (...arg: any[]) => any, delay = 500) {
  let flag = true
  return function (...args: any[]) {
    if (!flag) return
    flag = false
    setTimeout(() => {
      flag = true
      fn(...args)
    }, delay)
  }
}
